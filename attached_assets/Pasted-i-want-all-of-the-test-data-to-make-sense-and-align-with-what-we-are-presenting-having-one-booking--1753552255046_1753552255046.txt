import { sql } from 'drizzle-orm';
import { users, players, parentPlayers, futsalSessions, signups, payments, helpRequests, notificationPreferences } from './schema';
import { db } from './db';

// 1. Create parents & players
const parentIds = Array.from({ length: 10 }, (_, i) => `parent${i}`);
await db.insert(users).values(parentIds.map(id => ({
  id,
  email: `${id}@example.com`,
  firstName: `Parent${i}`,
  lastName: `User${i}`,
})));

let playerCount = 0;
for (const parentId of parentIds) {
  const numPlayers = Math.random() < 0.5 ? 1 : 2;
  for (let j = 0; j < numPlayers; j++) {
    const birthYear = 2025 - (8 + Math.floor(Math.random()*8));  // age 8–15
    const gender = Math.random()<0.5 ? 'boys' : 'girls';
    const pid = `player${playerCount++}`;
    await db.insert(players).values({
      id: pid,
      parentId,
      firstName: `Player${playerCount}`,
      lastName: `Test`,
      birthYear,
      gender,
      canAccessPortal: birthYear <= 2012,       // 13+ cutoff
      canBookAndPay: birthYear <= 2012,
    });
    await db.insert(parentPlayers).values({ parentId, playerId: pid });
  }
  // Notification prefs
  await db.insert(notificationPreferences).values({
    parentId,
    email: true,
    sms: Math.random()<0.5,
  });
}

// 2. Create sessions – Mon–Fri, 4 weeks
const startDate = new Date('2025-08-04'); // choose your month
for (let day = 0; day < 28; day++) {
  const date = new Date(startDate);
  date.setDate(startDate.getDate() + day);
  const wd = date.getDay();
  if (wd === 0 || wd === 6) continue;     // skip weekends

  const times = ['17:00', '18:45', '20:30'];
  for (let idx = 0; idx < times.length; idx++) {
    const [hh,mm] = times[idx].split(':').map(Number);
    const start = new Date(date); start.setHours(hh, mm);
    const end   = new Date(start); end.setMinutes(start.getMinutes()+90);

    const ageGroups = ['U8','U10','U12','U14','U15'];
    const genders   = ['boys','girls'];
    const ageGroup  = ageGroups[(day + idx) % ageGroups.length];
    const gender    = genders[(day + idx) % genders.length];

    await db.insert(futsalSessions).values({
      id: sql`gen_random_uuid()`,
      title: `${ageGroup} ${gender.charAt(0).toUpperCase()+gender.slice(1)} Session`,
      location: 'Sugar Sand Park, Boca Raton',
      ageGroup, gender,
      startTime: start,
      endTime: end,
      capacity: 12,
      priceCents: 1000,
      status: 'upcoming',
    });
  }
}

// 3. Signups & payments
const allPlayers = await db.select().from(players);
const allSessions = await db.select().from(futsalSessions);
for (const sess of allSessions) {
  // find eligible players
  const eligible = allPlayers.filter(p => {
    const age = new Date().getFullYear() - p.birthYear;
    return `U${age}` === sess.ageGroup && p.gender === sess.gender;
  });
  shuffle(eligible);
  const num = 8 + Math.floor(Math.random()*5); // 8–12
  for (let i = 0; i < num && i < eligible.length; i++) {
    const p = eligible[i];
    const signupId = sql`gen_random_uuid()`;
    await db.insert(signups).values({
      id: signupId,
      playerId: p.id,
      sessionId: sess.id,
      paid: true,
      createdAt: new Date(sess.startTime.getTime() - Math.random()*86400000),
    });
    await db.insert(payments).values({
      id: sql`gen_random_uuid()`,
      signupId,
      stripePaymentIntentId: `pi_${Math.random().toString(36).slice(2)}`,
      amountCents: 1000,
      paidAt: new Date(),
      createdAt: new Date(),
    });
  }
}

// 4. Help requests
await db.insert(helpRequests).values([
  { id: sql`gen_random_uuid()`, name: 'Alice Parent', email:'alice@example.com', note:'I can’t see my sessions', createdAt:new Date() },
  { id: sql`gen_random_uuid()`, name: 'Bob Parent',   email:'bob@example.com',   note:'Forgot password',       createdAt:new Date(), resolved:true },
  // …
]);
