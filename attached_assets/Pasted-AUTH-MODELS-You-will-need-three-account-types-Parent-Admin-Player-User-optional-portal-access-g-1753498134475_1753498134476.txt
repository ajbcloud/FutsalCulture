// frontend
const showSession = (player, session) => {
  const today = new Date().toDateString()
  const sessionDate = new Date(session.startTime).toDateString()
  const currentHour = new Date().getHours()

  return (
    session.ageGroup === calculateAgeGroup(player.birthYear) &&
    sessionDate === today &&
    currentHour >= 8 &&
    session.status === 'open'
  )
}
