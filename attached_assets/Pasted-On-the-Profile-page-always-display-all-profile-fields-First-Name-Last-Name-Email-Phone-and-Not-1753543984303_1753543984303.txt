<div className="space-y-6">
  {/* Profile Header */}
  <div className="flex justify-between items-center">
    <h2 className="text-2xl font-bold">Profile Information</h2>
    {isEditing ? (
      <div className="flex gap-2">
        <button onClick={saveProfile} className="bg-green-600 px-4 py-2 rounded">Save Changes</button>
        <button onClick={() => setIsEditing(false)} className="bg-zinc-700 px-4 py-2 rounded">Cancel</button>

  {/* Profile Fields */}

    {["firstName","lastName","email","phone"].map(field => (
      <div key={field} className="space-y-1">
        <label className="text-sm font-medium capitalize">
          {field === "phone" ? "Phone Number" : field.replace(/([A-Z])/g, " $1")}

          </div>
        )}
      </div>
    ))}
  </div>

  {/* Notification Preferences */}
  <div className="space-y-2">
    <h3 className="text-xl font-semibold">Notification Preferences</h3>
    {["emailReminder","smsReminder"].map(key => (
      <div key={key} className="flex justify-between items-center">
        <div>
          <p className="font-medium">
            {key === "emailReminder" ? "Email Reminders" : "SMS Reminders"}
          </p>
          <p className="text-sm text-zinc-400">
            Receive booking confirmations and session reminders via {key.includes("email") ? "email" : "SMS"}.
          </p>
        </div>
        <Toggle
          checked={formData[key]}
          disabled={!isEditing}
          onCheckedChange={val =>

          }
        />
      </div>
